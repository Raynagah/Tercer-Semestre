SELECT NUMRUN_CLI || ' - ' || DVRUN_CLI AS "RUN CLIENTE",
LOWER(PNOMBRE_CLI) || ' ' || INITCAP(SNOMBRE_CLI) || ' ' || APPATERNO_CLI || ' ' || APMATERNO_CLI AS "NOMBRE COMPLETO CLIENTE",
FECHA_NAC_CLI AS "FECHA NACIMIENTO" FROM CLIENTE 
--WHERE TO_CHAR(SYSDATE+1,'dd/mm')=TO_CHAR(FECHA_NAC_CLI,'dd/mm')-- 
WHERE '21/08'=TO_CHAR(FECHA_NAC_CLI,'dd/mm') 
ORDER BY APPATERNO_CLI;



--WHERE '21/08'=TO_CHAR(FECHA_NAC_CLI,'dd/mm'); Forma alternativa de hacerlo--

Para resolver el caso número 2 de la actividad 1.1:

SELECT NUMRUN_EMP || ' - ' || DVRUN_EMP AS "RUN EMPLEADO",
PNOMBRE_EMP || ' ' || SNOMBRE_EMP || ' ' || APPATERNO_EMP || ' ' || APMATERNO_EMP AS "NOMBRE COMPLETO EMPLEADO",
SUELDO_BASE,
TRUNC(SUELDO_BASE*0.00001,0) AS "PORCENTAJE MOVILIZACIÓN",
ROUND (SUELDO_BASE*TRUNC(SUELDO_BASE*0.00001)/100) AS "VALOR MOVILIZACIÓN"
FROM EMPLEADO ORDER BY "PORCENTAJE MOVILIZACIÓN" DESC;
