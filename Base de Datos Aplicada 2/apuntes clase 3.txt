SELECT * FROM EMPLEADO;
SELECT MAX(SALARIO) FROM EMPLEADO; --MOSTRAMOS EL SALARIO MÁXIMO DE LOS EMPLEADOS
SELECT MIN(SALARIO) FROM EMPLEADO; --MOSTRAMOS EL SALARIO MÍNIMO DE LOS EMPLEADOS
SELECT SUM(SALARIO) FROM EMPLEADO; --MOSTRAMOS LA SUMA DE LOS SALARIOS

SELECT ROUND(AVG(SALARIO)) FROM EMPLEADO; --MOSTRAMOS EL PROMEDIO DE LOS SALARIOS 

SELECT COUNT (RUN_EMP) FROM EMPLEADO; --CONTAMOS LA CANTIDAD DE RUNS DE LOS EMPLEADOS

SELECT * FROM EMPLEADO WHERE ID_ESTCIVIL=1; --MOSTRAMOS A LOS EMPLEADOS CUYO ESTADO CIVIL SEA IGUAL A 1

SELECT MAX(SALARIO) FROM EMPLEADO WHERE ID_ESTCIVIL=1; --MOSTRAMOS EL SALARIO MÁXIMO DE LOS EMPLEADOS CON ESTADO CIVIL 1

SELECT ID_ESTCIVIL,MAX(SALARIO),MIN(SALARIO),AVG(SALARIO) FROM EMPLEADO GROUP BY ID_ESTCIVIL; --MOSTRAMOS LA ID, EL SALARIO MÁXIMO, EL MÍNIMO Y EL PROMEDIO DE LOS EMPLEADOS AGRUPADOS POR ESTADO CIVIL

SELECT ID_ESTCIVIL,MAX(SALARIO) FROM EMPLEADO GROUP BY ID_ESTCIVIL HAVING MAX(SALARIO)>=1000000;

SELECT ID_ESTCIVIL,MAX(SALARIO) FROM EMPLEADO WHERE ID_ESTCIVIL>=2 GROUP BY ID_ESTCIVIL HAVING MAX(SALARIO)>=1000000; --PRIMERO FROM, LUEGO WHERE, LUEGO GROUP, LUEGO HAVING Y FINALMENTE ORDER BY EN ESE ORDEN DEBEN IR.

--OBTENER SALARIO MÁXIMO DE LOS EMPLEADOS DONDE SU NOMBRE COMIENZA CON N

SELECT MAX(SALARIO) FROM EMPLEADO WHERE SUBSTR(NOMBRE,1,1)='M';

SELECT *
FROM EMPLEADO
WHERE NOMBRE LIKE 'M%'; --MOSTRAMOS LOS NOMBRES QUE COMIENCEN POR EL CARACTER M Y AÑADIENDO % LE DECIMOS QUE NO NOS IMPORTA LO QUE TENGA DESPUÉS

SELECT *
FROM EMPLEADO
WHERE NOMBRE LIKE '%a'; --MOSTRAMOS LOS NOMBRES QUE TERMINEN POR EL CARACTER a Y AÑADIENDO % LE DECIMOS QUE NO NOS IMPORTA LO QUE TENGA ANTES


SELECT *
FROM EMPLEADO
WHERE NOMBRE LIKE '%ri%'; --MOSTRAMOS LOS NOMBRES QUE CONTENGAN ri SIN IMPORTARNOS LO QUE TENGA ANTES O DESPUÉS DE ESTOS

SELECT CARRERAID AS "IDENTIFICACION DE LA CARRERA",
COUNT (ALUMNOID) AS "TOTAL ALUMNOS MATRICULADOS",

'Le corresponden'||TO_CHAR(COUNT(ALUMNOID)*30200,'L999G999') || ' del presupuesto total asignado para publicidad' AS "MONTO POR PUBLICIDAD"
FROM ALUMNO
GROUP BY CARRERAID
ORDER BY COUNT(ALUMNOID) DESC;

SELECT TO_CHAR(100000,'L999G999G999') FROM DUAL; --DE ESTA FORMA LE DAMOS EL FORMATO DE LA MONEDA LOCAL 'L' Y LE INDICAMOS CON EL NÚMERO 9 QUE PUEDE CONTENER VALORES QUE VAN DESDE EL 0 AL 9

SELECT CARRERAID AS "CARRERA",
COUNT(ALUMNOID) AS "TOTAL DE ALUMNOS"
FROM ALUMNO
GROUP BY CARRERAID 
HAVING COUNT(ALUMNOID)>=5
ORDER BY "CARRERA";
